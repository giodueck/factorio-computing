.macro
@const f1 #1

.program
start:
    MOV R0 #0   ; zeroeth fib number
    MOV R1, f1   ; first fib number
; Line comment


    loop:
        ADD R2 R0 R1
        MOV R0 R1
        MOV R1 R2
        CALL subroutine
        JMP loop

subroutine:
    @rep 3
    ADD NIL @{0, 1} @{3, -1}
    MOV NIL @{0, 1}
    @end
    RETURN

finish:
